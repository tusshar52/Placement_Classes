#include <stdio.h>
// insertion sort

void insertionSort(int arr[], int n){
    int i, key, j;
    for (i = 1; i < n; i++){
        key = arr[i];
        j = i - 1;
        while (j >= 0 && arr[j] > key){
            arr[j + 1] = arr[j];
            j = j - 1;
        }
        arr[j + 1] = key;
    }
}

void display(int arr[], int n)
{
    int i;
    for (i = 0; i < n; i++)
        printf("%d \n", arr[i]);
}

int main()
{
    int arr[] = { 12, 17, 9 , 3, 55, 6 };
    int n = sizeof(arr) / sizeof(arr[0]);
 
    insertionSort(arr, n);
    display(arr, n);
 
    return 0;
}

//Output -

//3 
//6 
//9 
//12 
//17 
//55
